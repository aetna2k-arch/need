<script>

    let db;

    let auth;

    let user = null;

    const appId = typeof __app_id !== 'undefined' ? __app_id : 'mudan-survey-v4-active';

   

    // 初始化 UI

    async function init() {

        lucide.createIcons();

        renderOptions();

       

        // Firebase 初始化與身份驗證

        try {

            if (typeof __firebase_config !== 'undefined') {

                const config = JSON.parse(__firebase_config);

                firebase.initializeApp(config);

                db = firebase.firestore();

                auth = firebase.auth();

                // 核心修復：在執行任何 Firestore 操作前先進行驗證

                const initAuth = async () => {

                    if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                        await auth.signInWithCustomToken(__initial_auth_token);

                    } else {

                        await auth.signInAnonymously();

                    }

                };

                await initAuth();

                auth.onAuthStateChanged((u) => {

                    user = u;

                    console.log("Auth State Changed: ", u ? "Logged In" : "Logged Out");

                });

            }

        } catch (e) {

            console.error("Firebase Initialization Error:", e);

        }

        document.getElementById('feedback-form').onsubmit = handleSubmit;

        document.getElementById('admin-trigger').onclick = () => document.getElementById('login-modal').classList.remove('hidden');

    }

    function renderOptions() {

        const specs = ["生態解說", "歷史人文", "自然地理", "環境教育", "其他"];

        document.getElementById('spec-list').innerHTML = specs.map(s => `

            <div>

                <input type="checkbox" name="spec" value="${s}" id="s-${s}" class="sr-only checkbox-custom" onchange="toggleSpecOther(this)">

                <label for="s-${s}" class="block p-3 border-2 border-gray-100 rounded-xl text-center text-xs font-bold text-gray-400 cursor-pointer transition-all">${s}</label>

            </div>

        `).join('');

        const ratings = [

            { id: 'res', label: '場域教育資源豐富度' },

            { id: 'flow', label: '行程動線行走舒適度' },

            { id: 'food', label: '餐飲內容在地滿意度' }

        ];

        document.getElementById('rating-container').innerHTML = ratings.map(r => `

            <div>

                <p class="text-xs font-bold text-emerald-700 mb-3">${r.label}</p>

                <div class="flex gap-1">

                    ${[1,2,3,4,5].map(v => `

                        <label class="flex-1">

                            <input type="radio" name="score_${r.id}" value="${v}" required class="sr-only peer">

                            <div class="py-3 text-center border-2 border-gray-100 rounded-xl peer-checked:bg-emerald-700 peer-checked:text-white peer-checked:border-emerald-700 text-gray-300 font-black cursor-pointer">${v}</div>

                        </label>

                    `).join('')}

                </div>

            </div>

        `).join('');

        const prices = ["450元 (基本成本)", "600-750元 (專業補貼)", "800-1000元 (市場行情)", "其他"];

        document.getElementById('price-list').innerHTML = prices.map(p => `

            <label class="flex items-center gap-3 p-3 bg-white rounded-xl border-2 border-transparent hover:border-emerald-200 cursor-pointer">

                <input type="radio" name="price" value="${p}" required onchange="togglePriceOther(this.value)" class="w-4 h-4 accent-emerald-600">

                <span class="text-sm font-bold text-gray-600">${p}</span>

            </label>

        `).join('');

        const intents = ["願意帶團", "評估考慮", "暫無意願"];

        document.getElementById('intent-list').innerHTML = intents.map(i => `

            <label class="px-5 py-2 border-2 border-gray-100 rounded-full cursor-pointer has-[:checked]:bg-emerald-100 has-[:checked]:border-emerald-500 transition-all">

                <input type="radio" name="intent" value="${i}" required class="sr-only">

                <span class="text-xs font-bold text-gray-500">${i}</span>

            </label>

        `).join('');

    }

    function toggleSpecOther(el) {

        if(el.value === '其他') document.getElementById('spec-other-input').classList.toggle('hidden', !el.checked);

    }

    function togglePriceOther(v) {

        document.getElementById('price-other-input').classList.toggle('hidden', v !== '其他');

    }

    function closeLogin() { document.getElementById('login-modal').classList.add('hidden'); }

    async function handleSubmit(e) {

        e.preventDefault();

        if (!user) {

            alert("正在連線至雲端服務，請稍候再試...");

            return;

        }

        const btn = document.getElementById('submit-btn');

        btn.disabled = true;

        btn.innerText = "儲存中...";

        const fd = new FormData(e.target);

        const specs = Array.from(document.querySelectorAll('input[name="spec"]:checked')).map(c => c.value);

        if(specs.includes('其他')) specs.push("備註:"+document.getElementById('spec-other-input').value);

        const payload = {

            name: fd.get('user_name'),

            specs: specs,

            scores: {

                res: fd.get('score_res'),

                flow: fd.get('score_flow'),

                food: fd.get('score_food')

            },

            price: fd.get('price') === '其他' ? fd.get('price-other-input') : fd.get('price'),

            intent: fd.get('intent'),

            highlight: fd.get('highlight'),

            comment: fd.get('comment'),

            time: new Date().toISOString()

        };

        try {

            // 修正：使用規範要求的路徑

            const colRef = db.collection('artifacts').doc(appId).collection('public').doc('data').collection('responses');

            await colRef.add(payload);

            alert("感謝您的填寫！回饋已成功送出。");

            location.reload();

        } catch (err) {

            console.error("Submission error:", err);

            alert("儲存失敗，請檢查網路連線或稍後再試。");

            btn.disabled = false;

            btn.innerText = "確認送出";

        }

    }

    function doLogin() {

        if(document.getElementById('admin-pass').value === 'mudan') {

            closeLogin();

            document.getElementById('survey-container').classList.add('hidden');

            document.getElementById('admin-container').classList.remove('hidden');

            if(db && user) fetchRecords();

        } else { alert("密碼錯誤"); }

    }

    let records = [];

    function fetchRecords() {

        // 修正：使用正確路徑並加入錯誤處理回調

        const colRef = db.collection('artifacts').doc(appId).collection('public').doc('data').collection('responses');

        colRef.orderBy('time', 'desc').onSnapshot(snap => {

            records = snap.docs.map(d => d.data());

            renderRecords();

        }, error => {

            console.error("Snapshot error:", error);

            if (error.code === 'permission-denied') {

                console.log("嘗試重新連接中...");

            }

        });

    }

    function renderRecords() {

        document.getElementById('data-count').innerText = `目前共 ${records.length} 筆資料`;

        document.getElementById('entries-list').innerHTML = records.map(r => `

            <div class="bg-white p-4 rounded-2xl border shadow-sm">

                <div class="flex justify-between items-start mb-2">

                    <h4 class="font-black text-emerald-900">${r.name}</h4>

                    <span class="text-[10px] text-gray-400">${new Date(r.time).toLocaleDateString()}</span>

                </div>

                <div class="text-[11px] text-gray-500 space-y-1">

                    <p><b>專長：</b>${(r.specs || []).join(', ')}</p>

                    <p><b>評分：</b>餐:${r.scores.food} / 資:${r.scores.res} / 動:${r.scores.flow}</p>

                    <p><b>價格/意願：</b>${r.price} / ${r.intent}</p>

                    <p class="text-gray-700 mt-2 bg-gray-50 p-2 rounded-lg italic">"${r.highlight}"</p>

                </div>

            </div>

        `).join('');

    }

    function exportCSV() {

        if(!records.length) return;

        const headers = ["姓名", "專長", "餐飲評分", "資源評分", "動線評分", "建議價格", "意願", "印象", "回饋", "時間"];

        const rows = records.map(r => [

            r.name, (r.specs||[]).join('|'), r.scores.food, r.scores.res, r.scores.flow, r.price, r.intent, `"${r.highlight}"`, `"${r.comment}"`, r.time

        ]);

        const content = "\uFEFF" + [headers.join(","), ...rows.map(row => row.join(","))].join("\n");

        const blob = new Blob([content], {type: 'text/csv;charset=utf-8'});

        const link = document.createElement('a');

        link.href = URL.createObjectURL(blob);

        link.download = `牡丹問卷資料_${new Date().toLocaleDateString()}.csv`;

        link.click();

    }

    window.onload = init;

</script>
